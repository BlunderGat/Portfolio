#!/bin/bash

CONFIG_FILE="$1"
source "$CONFIG_FILE"


# Формируем имена файлов с датой
TIMESTAMP=$(date +"%Y.%m.%d_%H%M%S")
BACKUP_FILE="$BACKUP_PATH/$BACKUP_PREFIX.$TIMESTAMP.tar.gz"
LOG_FILE="$BACKUP_PATH/$BACKUP_PREFIX.$TIMESTAMP.log"
DUMP_FILE="$BACKUP_PATH/temp_$TIMESTAMP.dump"


# Функция логирования
log_message() {
    local message="$1"
    local timestamp=$(date +"%Y-%m-%d %H:%M:%S")
    echo "[$timestamp] [BACKUP] $message" | tee -a "$LOG_FILE"
}

mkdir -p "$BACKUP_PATH"


# Начало копирования
log_message "База: $DB_BASE | Хост: $DB_HOST | Пользователь: $DB_USER"
log_message "Папка бэкапов: $BACKUP_PATH"
log_message "Начало копирования"


# Проверка свободного места
AVAILABLE_SPACE=$(df "$BACKUP_PATH" | awk 'NR==2 {print $4}' 2>/dev/null || echo "0")
AVAILABLE_SPACE_MB=$((AVAILABLE_SPACE / 1024))

if [ "$AVAILABLE_SPACE" -lt 10000 ]; then
    log_message "Недостаточно свободного места"
    exit 1
fi

log_message "Свободное место: ${AVAILABLE_SPACE_MB}MB"


# Создаем дамп базы данных
export PGPASSWORD="$DB_PASS"
pg_dump -h "$DB_HOST" -U "$DB_USER" -F c -b -v "$DB_BASE" > "$DUMP_FILE" 2>> "$LOG_FILE"
DUMP_EXIT_CODE=$?

case $DUMP_EXIT_CODE in
    0)
        DUMP_SIZE=$(du -h "$DUMP_FILE" | cut -f1)
        log_message "Дамп создан, размер: $DUMP_SIZE"
        ;;
    1)
        log_message "Не удалось подключиться к PostgreSQL"
        rm -f "$DUMP_FILE"
        exit 1
        ;;
    2)
        log_message "Ошибка сервера PostgreSQL"
        rm -f "$DUMP_FILE"
        exit 1
        ;;
    *)
        log_message "Неизвестная ошибка, не удалось создать дамп"
        rm -f "$DUMP_FILE"
        exit 1
        ;;
esac


# Архивируем дамп
tar -czf "$BACKUP_FILE" -C "$BACKUP_PATH" "temp_$TIMESTAMP.dump" 2>> "$LOG_FILE"

if [ $? -eq 0 ]; then
    BACKUP_SIZE=$(du -h "$BACKUP_FILE" | cut -f1)
    log_message "Архив создан: $(basename "$BACKUP_FILE") ($BACKUP_SIZE)"
    
else
    log_message "Не удалось создать архив"
    rm -f "$DUMP_FILE"
    exit 1
fi

rm -f "$DUMP_FILE"


# Считаем сколько бэкапов
cd "$BACKUP_PATH" || exit 1
BACKUP_FILES_COUNT=$(ls -1 "$BACKUP_PREFIX".*.tar.gz 2>/dev/null | wc -l)
log_message "Копий: $BACKUP_FILES_COUNT / $BACKUP_COUNT"

# Если файлов больше, чем нужно
if [ "$BACKUP_FILES_COUNT" -gt "$BACKUP_COUNT" ] && [ "$BACKUP_COUNT" -gt 0 ]; then
    NEED_TO_DELETE=$((BACKUP_FILES_COUNT - BACKUP_COUNT))
    
    # Получаем список файлов, сортированных по времени (старые сверху)
    ls -1tr "$BACKUP_PREFIX".*.tar.gz 2>/dev/null | \
        grep -v "$(basename "$BACKUP_FILE")" | \
        head -"$NEED_TO_DELETE" | \
        while read -r OLD_FILE; do
            log_message "Превышен лимит копий, удаляем самую старую копию: $OLD_FILE"
            rm -f "$OLD_FILE"
        done
    
    BACKUP_FILES_COUNT=$(ls -1 "$BACKUP_PREFIX".*.tar.gz 2>/dev/null | wc -l)
    log_message "Копий: $BACKUP_FILES_COUNT / $BACKUP_COUNT"
fi


log_message "Копирование завершено"
exit 0


# Команды для терминала
# Для постоянного копирования		5 1 * * 1-5 /home/$(whoami)/test_task/backup-sql.linux.txt /home/$(whoami)/test_task/config.config >> /var/log/backup-sql.log 2>&1
# Запуск				            ./backup-sql.linux.txt config.config	
# Удалить копии				        rm -f ~/backups/sqlbackup.*
